1. To-Do List App

यह प्रोजेक्ट उपयोगकर्ताओं को कार्य जोड़ने, देखने और हटाने की सुविधा देता है। यह लूप्स, लिस्ट्स और कंडीशनल्स के अभ्यास के लिए उपयुक्त है।

tasks = []

def add_task():
    task = input("नया कार्य दर्ज करें: ")
    tasks.append(task)
    print(f"कार्य '{task}' जोड़ा गया।")

def view_tasks():
    if not tasks:
        print("कोई कार्य उपलब्ध नहीं है।")
    else:
        for i, task in enumerate(tasks, start=1):
            print(f"{i}. {task}")

def delete_task():
    view_tasks()
    try:
        task_num = int(input("हटाने के लिए कार्य संख्या दर्ज करें: ")) - 1
        removed_task = tasks.pop(task_num)
        print(f"कार्य '{removed_task}' हटाया गया।")
    except (IndexError, ValueError):
        print("अमान्य कार्य संख्या।")

def menu():
    while True:
        print("\n1. कार्य जोड़ें  2. कार्य देखें  3. कार्य हटाएं  4. बाहर निकलें")
        choice = input("अपना चयन दर्ज करें: ")
        if choice == '1':
            add_task()
        elif choice == '2':
            view_tasks()
        elif choice == '3':
            delete_task()
        elif choice == '4':
            print("To-Do List App से बाहर निकल रहे हैं।")
            break
        else:
            print("अमान्य चयन। कृपया पुनः प्रयास करें।")

menu()

2. Number Guessing Game

यह गेम उपयोगकर्ता से एक संख्या का अनुमान लगाने के लिए कहता है जो कंप्यूटर द्वारा यादृच्छिक रूप से चुनी जाती है। यह लूप्स और कंडीशनल्स के अभ्यास के लिए उपयुक्त है।

import random

def guessing_game():
    number_to_guess = random.randint(1, 100)
    attempts = 0
    print("1 से 100 के बीच की संख्या का अनुमान लगाएं।")

    while True:
        guess = int(input("अपना अनुमान दर्ज करें: "))
        attempts += 1
        if guess < number_to_guess:
            print("बहुत कम!")
        elif guess > number_to_guess:
            print("बहुत अधिक!")
        else:
            print(f"बधाई हो! आपने {attempts} प्रयासों में सही संख्या का अनुमान लगाया।")
            break

guessing_game()

3. Rock, Paper, Scissors Game

यह क्लासिक गेम उपयोगकर्ता और कंप्यूटर के बीच खेला जाता है। यह कंडीशनल्स और रैंडमनेस के अभ्यास के लिए उपयुक्त है।

import random

def play_game():
    choices = ["rock", "paper", "scissors"]
    computer_choice = random.choice(choices)
    player_choice = input("rock, paper, या scissors दर्ज करें: ").lower()

    if player_choice not in choices:
        print("अमान्य चयन। कृपया पुनः प्रयास करें।")
        return

    print(f"कंप्यूटर ने चुना: {computer_choice}")
    if player_choice == computer_choice:
        print("यह टाई है!")
    elif (player_choice == "rock" and computer_choice == "scissors") or \
         (player_choice == "paper" and computer_choice == "rock") or \
         (player_choice == "scissors" and computer_choice == "paper"):
        print("आप जीत गए!")
    else:
        print("आप हार गए।")

play_game()

4. Simple Chatbot

यह एक सरल चैटबॉट है जो उपयोगकर्ता के इनपुट के आधार पर पूर्वनिर्धारित उत्तर देता है। यह स्ट्रिंग हैंडलिंग और कंडीशनल्स के अभ्यास के लिए उपयुक्त है।

def chatbot_response(message):
    message = message.lower()
    if "hello" in message:
        return "नमस्ते! मैं आपकी कैसे मदद कर सकता हूँ?"
    elif "how are you" in message:
        return "मैं एक बॉट हूँ, इसलिए मैं हमेशा अच्छा हूँ!"
    elif "bye" in message:
        return "अलविदा! आपका दिन शुभ हो!"
    else:
        return "माफ़ कीजिए, मैं उस पर प्रतिक्रिया नहीं दे सकता। कुछ और पूछें?"

user_message = input("आप: ")
print("बॉट:", chatbot_response(user_message))

5. Email Slicer

यह प्रोजेक्ट एक ईमेल पते से उपयोगकर्ता नाम और डोमेन निकालता है। यह स्ट्रिंग मैनिपुलेशन के अभ्यास के लिए उपयुक्त है।

def email_slicer(email):
    username, domain = email.split('@')
    return username, domain

email = input("अपना ईमेल दर्ज करें: ").strip()
username, domain = email_slicer(email)
print(f"उपयोगकर्ता नाम: {username}, डोमेन: {domain}")

6. Password Generator

यह प्रोजेक्ट उपयोगकर्ता द्वारा निर्दिष्ट लंबाई के आधार पर एक मजबूत पासवर्ड उत्पन्न करता है। यह रैंडम और स्ट्रिंग मॉड्यूल के अभ्यास के लिए उपयुक्त है।

import random
import string

def password_generator(length):
    characters = string.ascii_letters + string.digits + string.punctuation
    password = ''.join(random.choice(characters) for _ in range(length))
    return password

length = int(input("पासवर्ड की लंबाई दर्ज करें: "))
print(f"उत्पन्न पासवर्ड: {password_generator(length)}")

7. Weather App (using API)

यह एप्लिकेशन OpenWeatherMap API का उपयोग करके किसी भी शहर के लिए वास्तविक समय का मौसम डेटा प्राप्त करता है। यह API हैंडलिंग और JSON डेटा के अभ्यास के लिए उपयुक्त है।

import requests

def get_weather(city):
    api_key = 'your_api_key'
    base_url = f"http://api.openweathermap.org/data/2.5/weather?q={city}&appid={api_key}&units=metric"
    response = requests.get(base_url)
    data = response.json()
    if data['cod'] == 200:
        main = data['main']
        weather = data['weather'][0]
        print(f"शहर: {city}")
        print(f"तापमान: {main['temp']}°C")
        print(f"मौसम: {weather['description']}")
    else:
        print("शहर नहीं मिला।")

city = input("शहर का नाम दर्ज करें: ")
get_weather(city)

8. Tic-Tac-Toe

यह दो खिलाड़ियों के बीच खेला जाने वाला क्लासिक गेम है। यह लूप्स, कंडीशनल्स और लिस्ट्स के अभ्यास के लिए उपयुक्त है।

def print_board(board):
    for row in board:
        print(" | ".join(row))
        print("-" * 5)

def check_winner(board, mark):
    for row in board:
        if all(s == mark for s in row):
            return True
    for col in range(3):
        if all(row[col] == mark for row in board):
            return True
    if all(board[i][i] == mark for i in range(3)) or all(board[i][2 - i] == mark for i in range(3)):
        return True
    return False

def tic_tac_toe():
    board = [[' ' for _ in range(3)] for _ in range(3)]
    current_player = "X"
    for _ in range(9):
        print_board(board)
        row, col = map(int, input(f"Player {current_player}, पंक्ति और स्तंभ दर्ज करें (0-2): ").split())
        if board[row][col] == ' ':
            board[row][col] = current_player
            if check_winner(board, current_player):
                print_board(board)
                print(f"Player {current_player} जीत गया!")
                return
            current_player = "O" if current_player == "X" else "X"
        else:
            print("सेल पहले से ही लिया गया है। पुनः प्रयास करें।")
    print_board(board)
    print("यह टाई है!")

tic_tac_toe()
